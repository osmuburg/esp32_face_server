# 1. Dirigirse a la carpeta del proyecto
cd ~/esp32_face_server

# 2. Enviar la IP del servidor a la ESP32-CAM
python3 send_ip_uart.py

NOTA: Solo ejecutar este comando si la IP del servidor no coincide con la IP que retorna el comando /ip en el chatboot.
Presione Ctrl + C luego que se envié la dirección IP

# 3. Ejecutar el servidor Flask
python app.py

# 4. Añadir imagenes de personas autorizadas desde la terminal
curl -X POST -F "name=PERSONA" -F "image=@foto.jpg" http://<ip_del_servidor>:5000/register

# 5. Verificar listado de personas con sus rostros autorizados
curl http://<ip_del_servidor>:5000/known

# 6. Probar reconocimiento con raw JPEG (sin la ESP)
curl -X POST --data-binary "@PERSONA.jpg" -H "Content-Type: image/jpeg" http://<ip_del_servidor>:5000/recognize
